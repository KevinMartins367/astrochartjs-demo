astrochartjs demo

This the demo of astrochartjs javascript library for drawing hindu astrological charts.

Javascript code


var options = {
        'title': ['Rasi', '14/04/2014 07:00AM', 'Erode, Tamil Nadu, India'],
        'showHouseNumbers' : true,
        'width': 540,
        'height': 400,
        'styleSheet':"http://rawgit.com/erajasekar/astrochartjs/master/dist/themes/astrosoft.css"
    };

var astroChart = new AstroChart("#chart1");
astroChart.draw({
    1: ["Su", "Ke"],
    3: ["Ju"],
    6: ["Ma", "Asc"],
    7: ["Mo", "Sa", "Ra"],
    11: ["Ve"],
    12: ["Me"]
}, options);
